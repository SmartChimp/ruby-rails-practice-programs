var VOWELS_ARRAY = ['a', 'e', 'i', 'o', 'u'];

        function isVowel(ch) {
            return (VOWELS_ARRAY.indexOf(ch) == -1) ? false : true;
        }

        function translateText(txtValue) {
            var txt_length = txtValue.length;
            var response = "";
            for (j = 0; j < txt_length; j++) {
                var ch = txtValue.charAt(j);
                var ascii_val = ch.charCodeAt(0);
                if ((ascii_val > 64 && ascii_val < 91) || (ascii_val > 96 && ascii_val < 123)) {
                    if (!isVowel(ch)) {
                        response += ch + 'o' + ch;
                    } else {
                        response += ch;
                    }
                } else {
                    response += ch;
                }
            }
            return response;
        }

        function execute(txtBoxElem) {
            txtValue = txtBoxElem.value;
            $('#result').text(translateText(txtValue));
        }
