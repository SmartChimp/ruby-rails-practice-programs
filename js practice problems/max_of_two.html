function findMax(num1, num2) {
		if(num1 > num2) {
			return num1;
		} else {
			return num2;
		}
	}
	function validate() {
		var num1 = $('#num1').val();
		var num2 = $('#num2').val();
		if(isNaN(num1)) {
			$('#result').text('first textbox contains invalid number.');	
			return;
		} 
		if(isNaN(num2)) {
			$('#result').text('second textbox contains invalid number.');	
			return;
		} 
		$('#result').text('result : ' + findMax(num1, num2));
	}
