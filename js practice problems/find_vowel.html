var VOWELS_ARRAY = ['a', 'e', 'i', 'o', 'u'];

        function isVowel(ch) {
            return (VOWELS_ARRAY.indexOf(ch.toLowerCase()) == -1)?false:true;
        }

        function validate() {
            var txtValue = $('#text-box-1').val();
            if(txtValue.length != 1) {
                $('#result').text('Invalid input. give single character.');
                return;
            }
            $('#result').text((isVowel(txtValue))? 'Its a vowel.': 'Its not a vowel.');
        }
