function reverseTxt(txtValue) {
            txt1 = '';
            txt2 = '';
            for(i=0,j=txtValue.length - 1; i<j; i++, j--) {
                txt1 = txtValue.charAt(i) + txt1;
                txt2 += txtValue.charAt(j);
            }
            if(i==j) {
                txt2 += txtValue.charAt(i);
            }
            return txt2+txt1;
        }
