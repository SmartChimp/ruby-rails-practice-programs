function sumArray(arrayValue) {
            var len = arrayValue.length;
            var result = 0;
            for(i=0,j=len-1; i<j; i++,j--){
                result += arrayValue[i]+arrayValue[j];
            }
            if(i==j) {
                result+=arrayValue[i];
            }
            return result;
        }
        function multipleArray(arrayValue) {
            var len = arrayValue.length;
            if(len > 0) {
                var result = 1;
                for(i=0,j=len-1; i<j; i++,j--){
                    result *= arrayValue[i]*arrayValue[j];
                }
                if(i==j) {
                    result*=arrayValue[i];
                }
                return result;
            }
            return 0;
        }
