text_hash = {
                "merry": "god",
                "christmas": "jul",
                "and": "och",
                "happy": "gott",
                "new": "nytt",
                "year": "år"
            }
        function translate(str) {
            var translatedString = "";

            str.toLowerCase().split(" ").map(function(currentValue, index, arr) {
                translatedString += text_hash[currentValue] + ' ';
            });

            return translatedString;
        }
