function maxOfThree(num1, num2, num3) {
			return (num1 > num2) ? ((num1>num3) ? num1 : num3) : ((num2 > num3) ? num2 : num3);
		}
		function validate() {
			num1 = parseInt($('#num1').val());
			num2 = parseInt($('#num2').val());
			num3 = parseInt($('#num3').val());
			if(isNaN(num1)) {
				$('#result').text('first textbox contains invalid number.');	
				return;
			} 
			if(isNaN(num2)) {
				$('#result').text('second textbox contains invalid number.');	
				return;
			} 
			if(isNaN(num3)) {
				$('#result').text('third textbox contains invalid number.');	
				return;
			} 
			$('#result').text('result : ' + maxOfThree(num1, num2, num3));
		}
