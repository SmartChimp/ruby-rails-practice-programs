function execute(txtBoxElement) {
            var stringValue = txtBoxElement.value;
            var i = 0;
            var frequencyArray = {};
            while(stringValue[i] != undefined) {
                if(stringValue[i] != ' ') {
                    (frequencyArray[stringValue[i]] === undefined)? frequencyArray[stringValue[i]] = 1 : frequencyArray[stringValue[i]] += 1;
                }
                i++;
            }
            var txt = '';
            for(key in frequencyArray) {
                txt += ' ['+key+','+frequencyArray[key]+']';
            }
            document.getElementById('result').innerHTML = 'Character frequencies are '+txt;
        }
